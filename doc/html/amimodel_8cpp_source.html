253 void AMIModel::gen_data(double *res_vec) {
254  unsigned long samples_written = 0;
255  while (samples_written++ < number_of_rows_) {
256  if (!(gen_data_cnt_++ % samples_per_bit_)) {
257  int feedback = (shift_reg_[PRBS_LEN - 1] + shift_reg_[PRBS_LEN - 2]) % 2;
258  for (auto i = PRBS_LEN - 1; i > 0; i--)
259  shift_reg_[i] = shift_reg_[i - 1];
260  shift_reg_[0] = feedback;
261  if (shift_reg_[PRBS_LEN - 1])
262  gen_data_last_ = 1.0;
263  else
264  gen_data_last_ = -1.0;
265  }
266  *res_vec++ = gen_data_last_;
267  }
268 }
